static void check_filename ( char *  filename)
static

Funzione privata che controlla la validita' del nome del file binario su cui si vuole salvare la partita.

Parametri
[in,out]filenamenome eventualmente aggiustato se non conforme a quanto aspettato.
{
D1(cerr<<"D1 check filename\n")
int x ;
for ( x = 0 ; x<MAXWORD ; x++ )
{
if (( filename[x] == '.' ) || ( filename[x] == '\0' ) )
break ;
}
if ( x < MAXWORD-5 )
{
filename[x] = '.' ;
filename[x+1] = 'd' ;
filename[x+2] = 'a' ;
filename[x+3] = 't' ;
filename[x+4] = '\0' ;
}
else
{
filename[MAXWORD-5] = '.' ;
filename[MAXWORD-4] = 'd' ;
filename[MAXWORD-3] = 'a' ;
filename[MAXWORD-2] = 't' ;
filename[MAXWORD-1] = '\0' ;
}
D10(cerr<<"D10 check filename\n")
}

void import_tiles_names ( )

Funzione che carica da file binario l' elenco dei nomi delle tessere. Per poi memorizzarlo in un array.

{
D1(cerr<<"D1 import tiles names\n")
if ( name != NULL )
delete[] name ;
name = new _string [TILES] ;
ifstream file ( F_TILES ) ;
//debuggare se apertura fallita
char temp ;
while ( file.peek() != '@' )
file.get ( temp ) ;
file.get ( temp ) ;
file.get ( temp ) ;
char buffer[MAXWORD] ;
for ( int i = 0, n = 0 ; i < TILES ; i++ )
{
file.get( buffer, MAXWORD, '\n' ) ;
n = file.gcount() ;
name[i].word = new char[n+1] ;
for ( int j = 0 ; j < n+1 ; j++ )
name[i].word[j] = buffer[j] ;
//debuggare presenza terminatore
while (file.peek() != '\n')
file.get ( temp ) ;
file.get ( temp ) ;
}
file.get ( temp ) ;
file.close () ;
D10(cerr<<"D10 import tiles names\n")
}
